import collections

string=input('Silahkan masukkan kata---> ')
kata=string.upper()

krata=kata.split(' ')
ambil=collections.Counter(krata).most_common()

for katas in ambil:
       print('Kalimat yang ditemukan: %s\t: Jumlah kata adalah: %d'%(katas[0],katas[1]))
       break

# kemudian mas arul membuat penampung kata itu displit menjadi perkata dengan variabel: krata 
# kemudian mas arul menggunakan salah satu collections yaitu Counter untuk menghitung kata yang sering muncul dalam variabel krata dengan nama variabel: ambil
# kemudian membuat variabel baru untuk mengambil dari variabel ambil
