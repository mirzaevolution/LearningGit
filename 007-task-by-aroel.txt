baris=int(input('Silahkan masukkan nilai N : ' ))
def aroel():
    if (baris>=3):
        for n in range(baris + 1, 0, -1):
            for j in range(0, n - 1):
                print("*", end=' ')
            print(" ")
    else:
        print('Nilai yang anda masukkan <3 silahkan ulangi')

aroel()
